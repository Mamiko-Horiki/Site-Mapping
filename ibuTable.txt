CREATE TABLE IF NOT EXISTS `Kihon_ibu` (
  `IBU_id` int(7) unsigned zerofill NOT NULL AUTO_INCREMENT,
  `IBU_site` varchar(7) NOT NULL DEFAULT '',
  `IBU_no` varchar(20) NOT NULL DEFAULT '',
  `IBU_ku` varchar(8) NOT NULL DEFAULT '',
  `IBU_grid` varchar(20) NOT NULL DEFAULT '',
  `IBU_iko` varchar(20) NOT NULL DEFAULT '',
  `IBU_syu` varchar(40) NOT NULL DEFAULT '',
  `IBU_syubiko` text NOT NULL,
  `PageNo` varchar(12) NOT NULL DEFAULT '',
  `IBU_seiri` varchar(20) NOT NULL DEFAULT '',
  `IBU_sozai` int(3) unsigned zerofill NOT NULL DEFAULT '000',
  `mono_iro` text NOT NULL,
  `mono_chose` text NOT NULL,
  `mono_long` float NOT NULL,
  `long_kesson` varchar(4) NOT NULL,
  `mono_width` float NOT NULL,
  `width_kesson` varchar(4) NOT NULL,
  `mono_thic` float NOT NULL,
  `thic_kesson` varchar(4) NOT NULL,
  `weight` float NOT NULL,
  `weight_kesson` varchar(5) NOT NULL,
  `zairyo_name` varchar(50) NOT NULL,
  `mono_jiki` varchar(20) NOT NULL DEFAULT '',
  `mono_zanzon` varchar(8) NOT NULL DEFAULT '',
  `mono_biko` text NOT NULL,
  `mono_biko1` text NOT NULL,
  `mono_biko2` text NOT NULL,
  `mono_biko3` text NOT NULL,
  PRIMARY KEY (`IBU_id`),
  KEY `mono_toroku` (`IBU_no`),
  KEY `IBU_site` (`IBU_site`),
  KEY `IBU_ku` (`IBU_ku`),
  KEY `IBU_iko` (`IBU_iko`)
) ENGINE=MyISAM  DEFAULT CHARSET=utf8 COMMENT='遺物一覧'

-- テーブルの構造 `SiteList`
--

CREATE TABLE IF NOT EXISTS `SiteList` (
  `SID_No` int(7) unsigned zerofill NOT NULL AUTO_INCREMENT,
  `BookName` varchar(100) NOT NULL DEFAULT '',
  `BookYomi` varchar(100) NOT NULL DEFAULT '',
  `FukuSyomei` varchar(100) NOT NULL DEFAULT '',
  `Kanji` varchar(10) NOT NULL DEFAULT '',
  `SirizuName` varchar(100) NOT NULL DEFAULT '',
  `SirizuKan` varchar(10) NOT NULL DEFAULT '',
  `HensyuMan` varchar(140) NOT NULL,
  `HensyuKikan` varchar(100) NOT NULL DEFAULT '',
  `HakoKikan` varchar(100) NOT NULL DEFAULT '',
  `HakoDay` varchar(8) NOT NULL DEFAULT '',
  `HakoID` varchar(8) NOT NULL DEFAULT '',
  `PostNo` varchar(8) NOT NULL DEFAULT '',
  `TelNo` varchar(13) NOT NULL DEFAULT '',
  `Address` varchar(100) NOT NULL DEFAULT '',
  `SiteName` varchar(100) NOT NULL DEFAULT '',
  `SiteYomi` varchar(100) NOT NULL DEFAULT '',
  `SiteAdd` varchar(100) NOT NULL DEFAULT '',
  `SiteAddYomi` varchar(200) NOT NULL DEFAULT '',
  `CityID` varchar(6) NOT NULL DEFAULT '',
  `SiteID` varchar(8) NOT NULL DEFAULT '',
  `W_Tokei` varchar(10) NOT NULL DEFAULT '',
  `W_Hokui` varchar(10) NOT NULL DEFAULT '',
  `J_Hokui` varchar(10) NOT NULL DEFAULT '',
  `J_Tokei` varchar(10) NOT NULL DEFAULT '',
  `ResDay` text NOT NULL,
  `ResMen` text NOT NULL,
  `ResGen` varchar(80) NOT NULL DEFAULT '',
  `Time` text NOT NULL,
  `Kind` text NOT NULL,
  `Gaiyo` text NOT NULL,
  `Tokki` text NOT NULL,
  `Yoyaku` text NOT NULL,
  `PDFfile` text NOT NULL,
  `PDFsize` varchar(6) NOT NULL,
  `Hizuke` date NOT NULL DEFAULT '2006-12-01',
  PRIMARY KEY (`SID_No`)
) ENGINE=MyISAM  DEFAULT CHARSET=utf8 COMMENT='遺跡一覧';
